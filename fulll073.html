الاقتباس النوراني - يوم الاربعين من شظظادة مولانا الامام الحسين ص ع - 20 صفر المظفر 1435 هـ - موضع - اديثثور

ا ج يه مولى كون؛؟ كه باونما داعي؛، جه ني خوشي حاصل كروو يه} سلام نا راستاؤ؛، سلام نا هر راسته اثث ني خوشي نا سبب ملسس؛، انسس كوئي اثث ني خوشي نسس موكي نسس سلام انسس امان نا راسته دهوندوا جائي - تو ككمراه تهئي جائي انسس ككمراهي ما اتهرتا ثثهرسس، تو دنيا ما بهي انسس مو ت ثثححهي بهي سلام نا راسته مول نا سبب ملسس؛، جه وقت بهوتراؤ راسته بهلاوسس؛، تو بيان ايم ايو؛ كه اذان دو، تو راسته ملي جائي؛ يعني مولى نسس ياد كرو، اهنو دامن ثثكرو، تو نعمة نا هر راسته ملي جاسسس
سيدي خانجي فير صاحب يه ديكهو ا سككلا قسم قسم ني خدمتو كرسس؛، انسس كتناك اهوي شاكلة سي خدمة كرسس؛ كه داعي ني نظر اهنا ثثر ثثرٌسس كه ا لوككو امثل خدمة كرسس؛، سيدي خانجي فير صاحب يه ديكهو كه منسس تو كوئي- رياء بغير خدمة كروي؛، كوئي رياء بغير خدمة كروي؛، ديكهاؤ بغير خدمة كروي؛، مولاي خانجي فيريه – حق نا داعي نسس جه ححيز نا ثثسند كراؤتو تهو اهنسس دور كروانسس عمل كيدو انسس مولى ني خوشي حاصل كيدي تو اهنسس كيوا كيوا سلام نا راسته ملا، جو سلاام نا راسته نه ملسس تو كظظاطط نسس كظظاطط نكلي جائي
