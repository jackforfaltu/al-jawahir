الاقتباس النوراني - يوم عرس الداعي الاجل سيدنا محمد برهان الدين رض - 16 ربيع الاول 1436 هـ ممبئي

مؤمنين ني جماعة! بسم الله الرحمن الرحيم ولله العزة ولرسوله وللمؤمنين، خدا انسس خدا نا رسول، مؤمنين واسطسس عزة؛، تين دن قبل ليلة السبت محترم Prime Minister Narendra Modi ني ملاقاة تهئي، يظظاطط ممبئي ما، Rajbhavan ما، باواجي صاحب مولى نو كرم انسس نظر اهنا ثثر هتي، اهني بهي ككهني ذكرو كيدي، باواجي صاحب مولى اهنسس شال عناية كيدي هتي، اهني بهي ذكر كيدي كه ميطط ا شال اجسس بهي استعمال كروطط ححهوطط، ثثححهي واتو ما كظظسس؛ كه ميطط ككهنا سنتو نسس ملو ححهوطط، ثثن سيدنا محمد بهران الدين صاحب جيوا كوئي ديكها نتهي، انسس ايم خاص ذكر ايم كيدي كه ميطط سيدنا صاحب يه منسس اهوا وقت ما كرامة اثثي انسس سظظارو اثثو كه جه قوت ما كوئي مارا سامنسس جووا تيار نظظتا
في اثناء الكلام يه كظظيو كه منسس اثث لوككو نا ساتهـ بهي وات كروا نو ككمسس؛، ام باواجي صاحب ني ذكر كيدي، سيدنا صاحب بهي، - تو ثثوتسس كظظسس كه منسس سيدنا صاحب نسس - ميطط - اهنا سي وات كروانو ككهنو ككمسس؛، ا ايك بسس وار كظظيو ثثوتسس كه منسس ككهنو ككمسس؛ سيدنا صاحب، تو ككمسس} نسس، شان نا هتا، اهوا، ححظظره اهوو، اثث ني وات اهوي هتي، ثثححهي همنسس بهي كظظسس كه اثث نا ا شيرواد ني منسس ضرورة؛
دعاءكروطط ححهوطط كه ا Modiji ني Leadership ما ملك ما امان امان قائم رهسس، هندوستان ثثروان ححرٌهسس انسس Bharat نو ثثرححم دنيا ما وكهنائي، ويثثار روزككار انسس Industries ما ديش ككهنو اككل ودهسس، ايم ماري اححها؛، ماري دعاء؛
