الاقتباس النوراني - يوم الاربعين من شظظادة مولانا الامام الحسين ص ع - 20 صفر المظفر 1435 هـ - موضع - اديثثور

اسس برادرو! ذره غور كرو! اثثنا حدود سي سكهامن لو، جه مولى ني زيارة يظظاطط كرئيسس؛ يظظاطط - يه كيوو راسته واضح كيدو كه كيوا بهي حالة اوسس، كيوا بهي حالة اوسس، مشكل زمان اوسس، - مككر اثث يه مولى نو دامن نه موكو، مولى نو در نه موكو، كتنا مهوتا صاحب مككر كظظاطط كهرا رهيا، مهوتا صاحب كتنا خانجي فير صاحب! مككر كظظاطط كهرا رهيا كه مولانا دروزه نا وهاطط، انسس ثثوري رات كهرا رهيا، ذره ما ححلا نه ككيا ام، ذره ما كه بس ام ححلا جائي كه ححلو ثثححهي - ححلا جاؤ - ايم نه ككيا، وهاطط نسس وهاطط، تهيرا، دل ما ايم كه مولى ناراض هوئي تو كظظاطط جاؤطط؟
ثثاححهو مكر ر كظظوطط ححهوطط كه دل ما ايم كه مولى ناراض هوئي تو كظظاطط جاؤطط؟ كيا راسته طرف جاؤوطط،؟ تو ا مثل مولى ني رضوان حاصل كروانسس عمل كيدو تو مولى يه نوازي ليدا ..
سيدي خانجي فير ثثر انسس اثث نا فرزند مولاي علي بهائي ثثر بهي ككهنا ظلم تهيا؛، يظظاطط لكك كه امر عظيم تهيو، تو مولاي علي بهائي- خانجي فير صاحب نا فرزند يه قيد تهيا، قيد ما بهي روزه كرسس اثث- قيد نا اندر تو روزه كرسس- بندككي كرسس، صبر كرسس، مؤمنين بهي سككلا عمل كرسس، ا دنيا نا حالات هوئي، تو ححهوتوا نو؛ نسس يظظاطط سي، يظظاطط سي ححهوضضوانو؛ نسس تو عمل كروو} جوئيسس نسس! صبر كروو} جوئيسس نسس،! صبر كروو} ثثرسس، صبر كروو} جوئيسس، حالات اوسس دنيا نا اندر، جه سمجهسس ته سمجهسس، جه نسس نه سمجهسس ته نسس سمجهسس
